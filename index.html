Challenge #3

Find the Biggest & Smallest Numbers

Write a function that returns the smallest and largest number in an array
The return value must be an array

Sample Input: [11, 4, 3, 7, 1, 8]


def min_max(array):
  min = array[0]
  max = array[0]
  for i in range(len(array)):
    if array[i] < min:
      min = array[i]
    if array[i] > max:
      max = array[i]
  return [min, max]


function minMax(array){
  let min = array[0];
  let max = array[0];
  for(const num in array){
    if(num < min)
      min = num;
    if(num > max)
      max = num;
  }
  return [min, max];
}


import java.util.ArrayList;

public ArrayList<Double> minMax(ArrayList<Double> array){
  double min = array.get(0);
  double max = array.get(0);
  for(int i = 0; i < array.size(); i++){
    if(array.get(i) < min)
      min = array.get(i);
    if(array.get(i) > max)
      max = array.get(i);
  }
  ArrayList<Double> minMaxArray = new ArrayList<Double>();
  minMaxArray.add(min);
  minMaxArray.add(max);
  return minMaxArray;
}

Challenge #5

Oddish or Evenish?

Create a function that determines whether a number is Oddish or Evenish. A number is Oddish if the sum of all of its digits is odd, and a number is Evenish if the sum of all of its digits is even. If a number is Oddish, return "Oddish". Otherwise, return "Evenish".

Sample Input: 387432


def oddish_or_evenish(num):
  total = 0
  for char in str(num):
    total += int(char)
  return "Evenish" if total % 2 == 0 else "Oddish"


function evenishOrOddish(num){
  let total = 0;
  for(const char of num.toString()){
    total += parseInt(char);
  }
  return total % 2 == 0? "Evenish" : "Oddish";
}


public String evenishOrOddish(int num){
  int total = 0;
  String numStr = num+"";
  for(int i = 0; i < numStr.length(); i++){
    total += Integer.parseInt(numStr.charAt(i)+"");
  }
  return total % 2 == 0? "Evenish" : "Oddish";
}

Challenge #6

Military Time

Write a function that receives the time in 12-hour AM/PM format and returns a string representation of the time in military (24-hour) format.
Note: 00:00:00 is midnight in military time

Sample Input: "12:32:05AM"


def convert_time(standardTime):
  h = int(standardTime[0:2])
  suffix = standardTime[-2:]
  if suffix.lower() == 'pm':
    h += 12
  if h == 24:
    h = 0;
  h = "0"+str(h) if h < 10 else str(h)
  return h + standardTime[2:-2];
  


function convertTime(standardTime){
  let h = parseInt(standardTime.substring(0, 2));
  let suffix = standardTime.substring(standardTime.length-2, standardTime.length);
  if(suffix.toLowerCase() == 'pm')
    h += 12;
  if(h == 24)
    h = 0;
  if(h < 10)
    h = "0"+h;
  return h+standardTime.substring(2, standardTime.length-2);
}


public String convertTime(String standardTime){
  int h = Integer.parseInt(standardTime.substring(0, 2));
  String suffix = standardTime.substring(standardTime.length()-2, standardTime.length());
  if(suffix.toLowerCase().equals("pm"))
    h += 12;
  if(h == 24)
    h = 0;

  if(h < 10)
     return "0"+h+standardTime.substring(2, standardTime.length()-2);
  else
    return h+standardTime.substring(2, standardTime.length()-2);
}

Challenge #7

Working 9-5

Write a function that calculates overtime and pay associated with overtime.
Working 9 to 5: regular hours
After 5 PM is overtime

Your function gets an array with 4 values:
   Start of working day, in 24-hour notation
   End of working day. (Same format)
   Hourly rate
   Overtime multiplier

Your function should spit out:
   $ + earned that day (rounded to the hundredths)

Sample Input: [12.75, 17.5, 11, 1.25]


def over_time(workDayInfo):
  hours_worked = workDayInfo[1] - workDayInfo[0]
  hourly_rate = workDayInfo[2]
  over_time_multiplier = workDayInfo[3]
  work_day_pay = hours_worked * hourly_rate
  over_time_pay = 0
  if hours_worked > 9:
    over_time_pay = (hours_worked-8)*(hourly_rate * over_time_multiplier)
    work_day_pay = (8 * hourly_rate)
  return "$"+str("%.2f" % round(work_day_pay + over_time_pay, 2))


function overTime(workDayInfo){
  const hoursWorked = workDayInfo[1] - workDayInfo[0];
  const hourlyRate = workDayInfo[2];
  const overTimeMultiplier = workDayInfo[3];
  let workDayPay = hourlyRate * hoursWorked;
  let overTimePay = 0;
  if(hoursWorked > 9){
    overTimePay = (hoursWorked - 8) * (hourlyRate * overTimeMultiplier);
    workDayPay = hourlyRate * 8;
  }
  return "$"+(workDayPay + overTimePay).toFixed(2);
}
